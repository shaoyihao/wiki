    题目：

    分析：将直方图抽象为一个数组，数组中的每一个元素 \(q_i\) 都可以求出一个“最小值区间”，每个“最小值区间”都对应着一个以 \(q_i\) 为高的极大矩形。\(n\) 个极大矩形中的面积最大值就是答案。

    #include <iostream>
#include <vector>
#include <stack>
using namespace std;
typedef pair<int, int> PII;
typedef long long LL;

const int N = 100010;
int l[N], r[N];

int main()
{
    int n;
    while (cin >> n, n)
    {
        vector<PII> h;
        for (int i = 0; i < n; i ++) 
        {
            int x;
            cin >> x;
            h.push_back({i, x});
        }

        stack<PII> s1;
        for (int i = 0; i < n; i ++)
        {
            while (!s1.empty() && s1.top().second >= h[i].second) s1.pop();
            if (!s1.empty()) l[i] = s1.top().first + 1;
            else l[i] = 0;
            s1.push(h[i]);
        }

        stack<PII> s2;
        for (int i = n - 1; i >= 0; i --)
        {
            while (!s2.empty() && s2.top().second >= h[i].second) s2.pop();
            if (!s2.empty()) r[i] = s2.top().first - 1;
            else r[i] = n - 1;
            s2.push(h[i]);
        }

        LL res = 0;
        for (int i = 0; i < n; i ++)
        {
            int height = h[i].second;
            res = max(res, (LL)height * (r[i] - l[i] + 1));
        }
        cout << res << endl;
    }
    return 0;
}

    #include <iostream>
#include <vector>
#include <stack>
using namespace std;
typedef pair<int, int> PII;

const int N = 1001;
int d[N][N], l[N], r[N];
char g[N][N];
int n, m;

int f(int h[])
{
    vector<PII> q;
    for (int i = 1; i <= m; i++) q.push_back({i, h[i]}); //将元素的下标与元素值绑定在一起，进行存储

    stack<PII> s1; 
    for (int i = 0; i < m; i++)
    {
        while (!s1.empty() && s1.top().second >= q[i].second) s1.pop();

        if (!s1.empty()) l[q[i].first] = s1.top().first + 1; 
        else l[q[i].first] = 1; 

        s1.push(q[i]);
    }

    stack<PII> s2; 
    for (int i = m - 1; i >= 0; i--)
    {
        while (!s2.empty() && s2.top().second >= q[i].second) s2.pop();

        if (!s2.empty()) r[q[i].first] = s2.top().first - 1; 
        else r[q[i].first] = m; 

        s2.push(q[i]);
    }

    int res = 0;
    for (int i = 1; i <= m; i++)
        res = max(res, h[i] * (r[i] - l[i] + 1));

    return res;
}

int main()
{
    cin >> n >> m;
    for (int i = 1; i <= n; i++)
        for (int j = 1; j <= m; j++)
        {
            cin >> g[i][j];
            if (g[i][j] == 'F') d[i][j] = d[i - 1][j] + 1;
            else d[i][j] = 0;
        }

    int res = 0;
    for (int i = 1; i <= n; i ++) res = max(res, f(d[i]));

    cout << res * 3;
    return 0;
}

    变式2¶

    题目：求红色方格可以构成的矩形总数。

    思路：将第 \(i\) 行视为底边，则转化为“直方图中的矩形数问题”。每一行都可以求出一个矩形数，将 \(n\)​ 行的所有矩形数相加，就是整个平面中的矩形数。

    对于一个直方图，如何求其中的矩形数呢？分别分析每一列，求出该列相关的矩形，再将所有列相关的矩形数相加，即得该直方图中的矩形数。

    定义第 \(i\) 列相关的矩形族为：第 \(i\) 列对应的直方图中的极大矩形中 包含第 \(i\) 列的所有矩形。左端点有 \(l\) 种选择，右端点有 \(r\) 种选择，高度有 \(h\) 种选择，因此第 \(i\) 列相关的矩形数为：\(l\times r\times h\)。

    #include <iostream>
#include <vector>
#include <stack>
using namespace std;
typedef pair<int, int> PII;

const int N = 1001;
int d[N][N], l[N], r[N];
char g[N][N];
int n, m;

long long f(int h[])
{
    vector<PII> q;
    for (int i = 1; i <= m; i++) q.push_back({i, h[i]}); 

    stack<PII> s1; // 单调递减栈，求第i个元素左边第一个小于等于它的元素，从而得出闭区间的左端点
    for (int i = 0; i < m; i++)
    {
        while (!s1.empty() && s1.top().second > q[i].second) s1.pop(); // 注意是小于号

        if (!s1.empty()) l[q[i].first] = s1.top().first + 1; 
        else l[q[i].first] = 1; 

        s1.push(q[i]);
    }

    stack<PII> s2; // 单调递减栈，求第i个元素右边第一个小于它的元素，从而得出闭区间的右端点
    for (int i = m - 1; i >= 0; i--)
    {
        while (!s2.empty() && s2.top().second >= q[i].second) s2.pop();

        if (!s2.empty()) r[q[i].first] = s2.top().first - 1; 
        else r[q[i].first] = m;

        s2.push(q[i]);
    }

    long long res = 0;
    for (int i = 1; i <= m; i++)
        res += h[i] * (i - l[i] + 1) * (r[i] - i + 1);

    return res;
}

int main()
{
    cin >> n >> m;
    for (int i = 1; i <= n; i++)
        for (int j = 1; j <= m; j++)
        {
            cin >> g[i][j];
            if (g[i][j] == '.') d[i][j] = d[i - 1][j] + 1;
            else d[i][j] = 0;
        }

    long long res = 0;
    for (int i = 1; i <= n; i++) res += f(d[i]);
    cout << res;

    return 0;
}
